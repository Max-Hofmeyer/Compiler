# Tokens:
* ID -> letter(letter or digit)
* NUMBER -> digit, optional_fraction, optional_exponent, number
* CHARLITERAL -> ''
* STRING -> ""
* RELOP -> ==, !=, <, <=, >=, >
* ADDOP -> +, -, ||
* MULOP -> *, /, %, &&
* ASSIGNOP -> =
* LPAREN -> (
* RPAREN -> )
* LCURLY -> {
* RCURLY -> }
* LBRACKET -> [
* RBRACKET -> ]
* COMMA -> , 
* SEMICOLON -> ;
* NOT -> !
* COLON -> :
